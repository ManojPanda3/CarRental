# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def mergeTwoLists(self, list1: Optional[ListNode], list2: Optional[ListNode]) -> Optional[ListNode]:
        def nodetoList(lists):
            data = []
            if lists==None: return data
            if lists.next!=None :
                data.extend(nodetoList(lists.next))
            print(lists.val,data)
            data.append(lists.val)
            return data
        d1,d2 = nodetoList(list1),nodetoList(list2)
        d1.extend(d2)
        d1 = sorted(d2)[::-1]
        prev = ListNode(d1[0])
        for i in range(1,len(d1)):
            prev = ListNode(next=prev,val=d1[i])
        return prev
